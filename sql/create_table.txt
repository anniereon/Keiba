-- マスタテーブルの作成

CREATE TABLE course (
    course_id VARCHAR(4) PRIMARY KEY,     -- コースID
    racecourse VARCHAR(20),           -- 競馬場
    surface_type VARCHAR(10),              -- グラウンド（芝・ダートなど）
    distance INT,                          -- 距離（メートルなど）
    direction VARCHAR(10)            -- 左右（left-handed / right-handed など）
);

CREATE TABLE horse (
    horse_id VARCHAR(10) PRIMARY KEY,   -- 馬ID（主キー）
    horse_name VARCHAR(50)             -- 馬の名前
);

CREATE TABLE jockey (
    jockey_id VARCHAR(5) PRIMARY KEY,   -- 騎手ID（主キー）
    jockey_name VARCHAR(50)             -- 騎手名
);

CREATE TABLE bet_type (
    bet_type_id VARCHAR(3) PRIMARY KEY,    -- 馬券種ID（例: WIN, EXA）
    bet_type_name VARCHAR(10)               -- 馬券種名（例: 単勝, 馬単）
);

CREATE TABLE weather (
    weather_id VARCHAR(1) PRIMARY KEY,
    weather_name VARCHAR(10)
);

CREATE TABLE track_condition (
    track_condition_id VARCHAR(1) PRIMARY KEY,
    track_condition_name VARCHAR(10)
);

CREATE TABLE style (
    style_id VARCHAR(1) PRIMARY KEY,
    style_name VARCHAR(10)
);

-- トランザクションテーブルの作成

CREATE TABLE race (
    race_id VARCHAR(9) PRIMARY KEY,
    race_date DATE,
    num_horses INT,
    course_id VARCHAR(4),
    race_number INT,
    weather_id VARCHAR(1),
    track_condition_id VARCHAR(1),
    note TEXT,

    FOREIGN KEY (course_id) REFERENCES course(course_id),
    FOREIGN KEY (weather_id) REFERENCES weather(weather_id),
    FOREIGN KEY (track_condition_id) REFERENCES track_condition(track_condition_id)
);

CREATE TABLE race_detail (
    race_id VARCHAR(9),
    horse_id VARCHAR(10),
    jockey_id VARCHAR(5),
    horse_number INT,
    frame_number INT,
    style_id VARCHAR(1),
    odds DECIMAL(5,1),
    popularity INT,
    finish_rank INT,
    time_index INT,
    note TEXT,
    PRIMARY KEY (race_id, horse_number),
    FOREIGN KEY (race_id) REFERENCES race(race_id),
    FOREIGN KEY (horse_id) REFERENCES horse(horse_id),
    FOREIGN KEY (jockey_id) REFERENCES jockey(jockey_id),
    FOREIGN KEY (style_id) REFERENCES style(style_id)
);

-- サマリテーブルの作成

CREATE TABLE race_statistics (
    stat_id BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    course_id VARCHAR(4),
    num_horses INT,
    race_number INT,
    weather_id VARCHAR(1),
    horse_number INT,
    frame_number INT,
    style_id VARCHAR(1),
    sample_size INT,
    num_place INT,
    num_win INT,
    FOREIGN KEY (course_id) REFERENCES course(course_id),
    FOREIGN KEY (weather_id) REFERENCES weather(weather_id),
    FOREIGN KEY (style_id) REFERENCES style(style_id)
);
